--- Block statement
### ts
{
   1 + 1;
}

### c
{
1 + 1;
};

--- While statement
### ts
while (i > 10) {
i++;
}

### c
while (i > 10) {
i++;
};

--- While statement without brackets
### ts
while (i > 10)
i++;

### c
while (i > 10) i++;

--- Break statement
### ts
while (i > 10) {
break;
}

### c
while (i > 10) {
break;
};

--- Continue statement
### ts
while (i > 10) {
continue;
}

### c
while (i > 10) {
continue;
};

--- If statement with if
### ts
if (i > 10) {
i = 3;
}

### c
if (i > 10) {
i = 3;
};

--- If statement with if, else
### ts
if (i > 10) {
i = 3;
} else {
i = 4;
}

### c
if (i > 10) {
i = 3;
} else {
i = 4;
};

--- If statement with if, else if, else
### ts
if (i > 10) {
i = 3;
} else if (i < 20) {
i = 4;
} else {
i = 5;
}

### c
if (i > 10) {
i = 3;
} else if (i < 20) {
i = 4;
} else {
i = 5;
};

--- If statement without brackets
### ts
if (i > 10)
i = 3;

### c
if (i > 10) i = 3;

--- For statement
### ts
for (let n:integer = 0; n < 9; n++) {
i = i + 1;
}

### c
for (int32_t n = 0; n < 9; n++;) {
i = i + 1;
};

--- For statement without init, condition and update
### ts
for (;;) {
i = i + 1;
}

### c
for (; ; ;) {
i = i + 1;
};
